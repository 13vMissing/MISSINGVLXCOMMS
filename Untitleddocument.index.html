    function sortCommissions(order) {

        commissionData.sort((a, b) => new Date(a.created_at) - new Date(b.created_at));

      } else if (sortOrder === 'status') {

        const statusOrder = { 'pending': 0, 'accepted': 1, 'finished': 2, 'declined': 3 };

        commissionData.sort((a, b) => (statusOrder[a.status] || 99) - (statusOrder[b.status] || 99));

      }

      renderCommissionList();

    }

    function updateCommissionProgress(backendId, action) {

      const commission = commissionData.find(c => c.__backendId === backendId);

      if (!commission) return;

     

      const currentIndex = progressStages.indexOf(commission.progress_stage || 'Requested');

      let newStage = commission.progress_stage || 'Requested';

     

      if (action === 'next' && currentIndex < progressStages.length - 1) {

        newStage = progressStages[currentIndex + 1];

      }

     

      updateCommissionField(backendId, 'progress_stage', newStage);

    }

   

    async function updateCommissionField(backendId, field, value) {

      const commission = commissionData.find(c => c.__backendId === backendId);

      if (!commission) return;

     

      const updated = { ...commission, [field]: value };

      if (field === 'status' && value === 'finished') {

        updated.progress_stage = 'Finished';

      }

     

      const result = await window.dataSdk.update(updated);

      if (!result.isOk) showPopup('Failed to update! 😿');

    }

    function sortCommissions(order) {

      sortOrder = order;

      if (sortOrder === 'newest') {

        commissionData.sort((a, b) => new Date(b.created_at) - new Date(a.created_at));

      } else if (sortOrder === 'oldest') {

        commissionData.sort((a, b) => new Date(a.created_at) - new Date(b.created_at));

      } else if (sortOrder === 'status') {

        const statusOrder = { 'pending': 0, 'accepted': 1, 'finished': 2, 'declined': 3 };

        commissionData.sort((a, b) => (statusOrder[a.status] || 99) - (statusOrder[b.status] || 99));

      }

      renderCommissionList();

    }

    function updateCommissionProgress(backendId, action) {

      const commission = commissionData.find(c => c.__backendId === backendId);

      if (!commission) return;

     

      const currentIndex = progressStages.indexOf(commission.progress_stage || 'Requested');

      let newStage = commission.progress_stage || 'Requested';

     

      if (action === 'next' && currentIndex < progressStages.length - 1) {

        newStage = progressStages[currentIndex + 1];

      }

     

      updateCommissionField(backendId, 'progress_stage', newStage);

    }

   

    async function updateCommissionField(backendId, field, value) {

      const commission = commissionData.find(c => c.__backendId === backendId);

      if (!commission) return;

     

      const updated = { ...commission, [field]: value };

      if (field === 'status' && value === 'finished') {

        updated.progress_stage = 'Finished';

      }

     

      const result = await window.dataSdk.update(updated);

      if (!result.isOk) showPopup('Failed to update! 😿');

    }

   

    async function updateCommissionStatus(backendId, newStatus) {

      await updateCommissionField(backendId, 'status', newStatus);

    }
